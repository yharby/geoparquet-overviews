# geoparquet-overviews v0.2.0 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship overviews draft 0.2.0, dual GeoParquet 1.1 plus native GEOMETRY types with automatic row-group geospatial statistics, per-level byte ranges and extents in the footer, an opt-in no-bbox profile, viewer support for all of it plus a version dropdown fed by a hosted versions.json manifest, versioned test data on source.coop (v0.1.0 copies, v0.2.0 regenerated), and a benchmark comparing the two.

**Architecture:** The converter keeps its single-pipeline shape in `convert.py`. Native types come from wrapping the two WKB columns in the `geoarrow.wkb` extension type, which pyarrow 21+ converts to the Parquet GEOMETRY logical type with automatic GeospatialStatistics on write. Per-level byte ranges require knowing file offsets before the footer is written, solved by tracking `sink.tell()` per row group and injecting the `overviews` key late through `ParquetWriter.add_key_value_metadata`. The viewer gains a native-stats fallback for row-group bboxes when the covering column is absent, and a manifest-driven version dropdown. Data operations use the aws cli with the `sc-iam` profile against the real bucket `us-west-2.opendata.source.coop`.

**Tech Stack:** Python (pyarrow 24, shapely, click, new dep geoarrow-pyarrow), TypeScript (Lit, deck.gl 9.3.6, MapLibre, hyparquet 1.26.2, vitest), aws cli.

```python
def _unwrap_extension(table: pa.Table, name: str) -> pa.Table:
    """Replace an extension-typed column (geoarrow.wkb from a previous native
    run) with its plain binary storage so the decode path sees WKB bytes."""
    if name not in table.column_names:
        return table
    col = table.column(name)
    if not isinstance(col.type, pa.ExtensionType):
        return table
    storage = pa.chunked_array([c.storage for c in col.chunks])
    return table.set_column(table.column_names.index(name), name, storage)
```

```python
    table: pa.Table, dst: str, plan: list[int], kv: dict[str, str],
    opts: ConvertOptions, late_kv=None,
) -> None:
    ...  # metadata filtering and column option lists unchanged
    with open(dst, "wb") as sink:
        writer = pq.ParquetWriter(
            sink, table.schema,
            compression="zstd",
            compression_level=opts.compression_level,
            write_page_index=True,
            data_page_size=opts.page_size_kb * 1024,
            use_dictionary=dict_cols or False,
            use_byte_stream_split=byte_split,
            write_statistics=stats_cols,
            sorting_columns=sorting,
        )
        rg_ranges: list[tuple[int, int]] = []
        offset = 0
        for rows in plan:
            start = sink.tell()
            writer.write_table(table.slice(offset, rows), row_group_size=rows)
            rg_ranges.append((start, sink.tell()))
            offset += rows
        if late_kv is not None:
            writer.add_key_value_metadata(late_kv(rg_ranges))
        writer.close()
```

- [ ] **Step 3: Implement `manifest.ts`**

```typescript
import type { FilePreset } from './presets';

// The hosted test-data catalog. One JSON file at the prefix root lists every
// published data version and its datasets, so the viewer can offer a version
// dropdown and byte-cost comparisons across converter releases without a
// rebuild. Fetch failures degrade to the built-in presets, never to an error.
export interface ManifestDataset {
  id: string;
  label: string;
  path: string;
}

export interface ManifestVersion {
  version: string;
  prefix: string;
  spec?: string;
  datasets: ManifestDataset[];
}

export interface VersionManifest {
  latest: string;
  base: string;
  versions: ManifestVersion[];
}

export const MANIFEST_URL = 'https://data.source.coop/youssef-harby/geoparquet-overviews/versions.json';

function isDataset(d: unknown): d is ManifestDataset {
  const x = d as ManifestDataset;
  return !!x && typeof x.id === 'string' && typeof x.label === 'string' && typeof x.path === 'string';
}

function isVersion(v: unknown): v is ManifestVersion {
  const x = v as ManifestVersion;
  return (
    !!x &&
    typeof x.version === 'string' &&
    typeof x.prefix === 'string' &&
    Array.isArray(x.datasets) &&
    x.datasets.every(isDataset)
  );
}

export function parseManifest(raw: unknown): VersionManifest | null {
  const m = raw as VersionManifest;
  if (!m || typeof m.latest !== 'string' || typeof m.base !== 'string') return null;
  if (!Array.isArray(m.versions) || m.versions.length === 0 || !m.versions.every(isVersion)) return null;
  return { latest: m.latest, base: m.base.replace(/\/$/, ''), versions: m.versions };
}

export function presetsForVersion(manifest: VersionManifest, version: string): FilePreset[] {
  const entry = manifest.versions.find((v) => v.version === version);
  if (!entry) return [];
  return entry.datasets.map((d) => ({
    id: d.id,
    label: d.label,
    url: `${manifest.base}/${entry.prefix}/${d.path}`,
  }));
}

export async function fetchManifest(url = MANIFEST_URL): Promise<VersionManifest | null> {
  try {
    const res = await fetch(url);
    if (!res.ok) return null;
    return parseManifest(await res.json());
  } catch {
    return null;
  }
}
```

- [ ] **Step 2: The sweep**

`viewer/bench/compare.bench.ts`. It plans reads for a ladder of viewports per dataset per version and sums the compressed bytes of the planned geometry chunks, which is what the viewer would fetch at row-group granularity:

```typescript
import { describe, expect, it } from 'vitest';
import { asyncBufferFromUrl, parquetMetadataAsync } from 'hyparquet';
import { readGeoParquetMetadata } from '../src/data/metadata';
import { detectLayout } from '../src/data/layout';
import type { Bbox } from '../src/geo/aoi';

const BASE = 'https://data.source.coop/youssef-harby/geoparquet-overviews';
const DATASETS = [
  'sample.parquet',
  'big_sample.parquet',
  'nls_rakennus_overviews.parquet',
  'overture-tokyo/buildings.parquet',
  'overture-tokyo/segments.parquet',
  'overture-tokyo/pois.parquet',
];
// Viewport ladder as fractions of the dataset extent, with a zoom that a map
// showing that fraction of the world-side extent would sit at.
const VIEWPORTS = [
  { name: 'full extent', frac: 1, zoom: 5 },
  { name: 'region', frac: 1 / 8, zoom: 9 },
  { name: 'city', frac: 1 / 64, zoom: 12 },
  { name: 'district', frac: 1 / 512, zoom: 15 },
];

function centered(bbox: Bbox, frac: number): Bbox {
  const cx = (bbox.xmin + bbox.xmax) / 2;
  const cy = (bbox.ymin + bbox.ymax) / 2;
  const hw = ((bbox.xmax - bbox.xmin) * frac) / 2;
  const hh = ((bbox.ymax - bbox.ymin) * frac) / 2;
  return { xmin: cx - hw, ymin: cy - hh, xmax: cx + hw, ymax: cy + hh };
}

async function sweep(url: string) {
  const file = await asyncBufferFromUrl({ url });
  const raw = await parquetMetadataAsync(file);
  const meta = readGeoParquetMetadata(raw as never);
  const strategy = detectLayout(meta);
  const boxes = meta.rowGroups.map((rg) => rg.bbox).filter((b): b is Bbox => b !== null);
  const extent = boxes.reduce(
    (a, b) => ({
      xmin: Math.min(a.xmin, b.xmin),
      ymin: Math.min(a.ymin, b.ymin),
      xmax: Math.max(a.xmax, b.xmax),
      ymax: Math.max(a.ymax, b.ymax),
    }),
    { xmin: Infinity, ymin: Infinity, xmax: -Infinity, ymax: -Infinity },
  );
  const rows: Record<string, unknown>[] = [];
  for (const vp of VIEWPORTS) {
    const plan = strategy.planRead(centered(extent, vp.frac), vp.zoom);
    let bytes = 0;
    for (const idx of plan.indices) {
      const rg = raw.row_groups[idx];
      const chunk = rg.columns.find(
        (c) => c.meta_data && c.meta_data.path_in_schema.join('.') === plan.column,
      );
      bytes += Number(chunk?.meta_data?.total_compressed_size ?? 0);
    }
    rows.push({ viewport: vp.name, column: plan.column, rowGroups: plan.indices.length, mb: (bytes / 1e6).toFixed(2) });
  }
  return rows;
}

describe('v0.1.0 vs v0.2.0 planned read bytes', () => {
  for (const ds of DATASETS) {
    it(ds, async () => {
      const results: Record<string, unknown> = {};
      for (const version of ['v0.1.0', 'v0.2.0']) {
        results[version] = await sweep(`${BASE}/${version}/${ds}`);
      }
      // eslint-disable-next-line no-console
      console.log(JSON.stringify({ dataset: ds, results }, null, 2));
      expect(results).toBeTruthy();
    });
  }
});
```

(Adjust the two hyparquet call signatures to the installed 1.26.2 API while implementing, `asyncBufferFromUrl` takes `{url}` and `parquetMetadataAsync` returns the FileMetaData that `readGeoParquetMetadata` already consumes via `file-cache`, mirror how `src/data/file-cache.ts` calls them.)
